28// ---------------------------------------------------------------------------
29// Escape sink — a tiny interface type-erased on ValueType only (not on the
30// outer receiver). This lets escape_factory/escape_sender depend solely on
31// ValueType, so call_cc_sender can name the inner sender type (and thus derive
32// its completion signatures) without knowing the outer receiver.
33// ---------------------------------------------------------------------------
34template <class ValueType>
35struct escape_sink {
36 virtual void do_escape(ValueType) noexcept = 0;
37
38 protected:
39 ~escape_sink() = default;
40};
41
42// ---------------------------------------------------------------------------
43// Shared state — the synchronization node of a call_cc block. Holds the outer
44// receiver and an inplace_stop_source for downward cancellation. It does NOT
45// complete the outer receiver on escape; instead an escape stashes its value
46// here and requests stop, and the *inner operation's single completion* is the
47// sole point that completes the outer receiver (drain-then-complete). This
48// guarantees no inner work is still in flight when the outer completes.
49// ---------------------------------------------------------------------------
50template <class OuterReceiver, class ValueType>
51struct call_cc_shared_state : escape_sink<ValueType> {
52 OuterReceiver outer_receiver;
53 ex::inplace_stop_source stop_source;
54 std::atomic<bool> escape_claimed{false}; // one-shot winner latch
55 std::atomic<bool> escaped{false}; // publishes escape_value
56 std::optional<ValueType> escape_value;
57
58 explicit call_cc_shared_state(OuterReceiver &&rcvr)
59 : outer_receiver(std::move(rcvr)) {}
60
61 // Called (possibly from a worker thread) when an escape sender starts.
62 void do_escape(ValueType v) noexcept override {
63 if (!escape_claimed.exchange(true, std::memory_order_acq_rel)) {
64 escape_value.emplace(std::move(v));
65 escaped.store(true, std::memory_order_release); // publish value
66 stop_source.request_stop(); // cancel siblings
67 }
68 }
69
70 bool has_escaped() const noexcept {
71 return escaped.load(std::memory_order_acquire);
72 }
73};

75// ---------------------------------------------------------------------------
76// Escape sender — the reified captured continuation. When started it stashes
77// the escaped value into the sink and requests stop, then completes its OWN
78// local receiver with set_stopped() so the enclosing graph cancels and drains.
79// The escaped value is delivered to the outer receiver later, by the inner
80// operation's single completion (see inner_receiver). Hence it advertises
81// set_stopped_t().
82// ---------------------------------------------------------------------------
83template <class ValueType, class LocalReceiver>
84struct escape_op_state {
85 using operation_state_concept = ex::operation_state_tag;
86
87 escape_sink<ValueType> *sink;
88 ValueType value;
89 LocalReceiver local_receiver;
90
91 void start() & noexcept {
92 sink->do_escape(std::move(value));
93 ex::set_stopped(std::move(local_receiver));
94 }
95};

188// ---------------------------------------------------------------------------
189// inner_env_t — the environment the inner sender sees. Built from beman's own
190// env-composition utilities (the same pattern beman's let adaptor uses): a
191// prop overriding get_stop_token with the local inplace_stop_token, joined
192// over a forwarding view of the outer environment. Used both at runtime and
193// for completion-signature derivation, so the two always agree.
194// ---------------------------------------------------------------------------
195template <class OuterEnv>
196using inner_env_t = decltype(ex::detail::join_env(
197 ex::prop{ex::get_stop_token, std::declval<ex::inplace_stop_token>()},
198 ex::detail::fwd_env(std::declval<OuterEnv>())));
199
200// ---------------------------------------------------------------------------
201// Inner receiver — intercepts the user's inner computation. Its environment
202// injects the local stop token. Its completion is the SOLE point that
203// completes the outer receiver: if an escape was claimed, deliver the stashed
204// value via set_value; otherwise forward the inner sender's own completion.
205// ---------------------------------------------------------------------------
206template <class SharedState>
207struct inner_receiver {
208 using receiver_concept = ex::receiver_tag;
209
210 SharedState *shared_state;
211
212 template <class Fallback>
213 void finish(Fallback &&fallback) noexcept {
214 if (shared_state->has_escaped()) {
215 ex::set_value(std::move(shared_state->outer_receiver),
216 std::move(*shared_state->escape_value));
217 } else {
218 std::forward<Fallback>(fallback)();
219 }
220 }
221
222 template <class... Args>
223 void set_value(Args &&...args) && noexcept {
224 finish([&] {
225 ex::set_value(std::move(shared_state->outer_receiver),
226 std::forward<Args>(args)...);
227 });
228 }
229
230 template <class Error>
231 void set_error(Error &&err) && noexcept {
232 finish([&] {
233 ex::set_error(std::move(shared_state->outer_receiver),
234 std::forward<Error>(err));
235 });
236 }
237
238 void set_stopped() && noexcept {
239 finish(
240 [&] { ex::set_stopped(std::move(shared_state->outer_receiver)); });
241 }
242
243 auto get_env() const noexcept {
244 return ex::detail::join_env(
245 ex::prop{ex::get_stop_token, shared_state->stop_source.get_token()},
246 ex::detail::fwd_env(ex::get_env(shared_state->outer_receiver)));
247 }
248};

250// ---------------------------------------------------------------------------
251// stop_forwarder — a trivial callable that bridges outer cancellation to the
252// local inplace_stop_source. Replaces std::function<void()> in the stop
253// callback so that zero heap allocation is structurally guaranteed.
254// ---------------------------------------------------------------------------
255template <class SharedStateType>
256struct stop_forwarder {
257 SharedStateType *shared_state;
258 void operator()() const noexcept {
259 shared_state->stop_source.request_stop();
260 }
261};

263// ---------------------------------------------------------------------------
264// Outer operation state — weaves the pieces together. Owns the shared state by
265// value (no heap allocation), then connects and starts the user's inner sender.
266//
267// The inner operation state is held in manual union storage rather than a
268// std::optional because operation states are immovable; placement-new from the
269// connect() prvalue constructs it in place via guaranteed copy elision.
270// ---------------------------------------------------------------------------
271template <class OuterReceiver, class F, class ValueType>
272struct call_cc_op_state {
273 using operation_state_concept = ex::operation_state_tag;
274
275 using SharedStateType = call_cc_shared_state<OuterReceiver, ValueType>;
276 using InnerSenderType =
277 std::invoke_result_t<F &, escape_factory<ValueType>>;
278 using InnerReceiverType = inner_receiver<SharedStateType>;
279 using InnerOpStateType =
280 ex::connect_result_t<InnerSenderType, InnerReceiverType>;
281
282 using OuterStopToken = decltype(ex::get_stop_token(
283 ex::get_env(std::declval<const OuterReceiver &>())));
284 using StopCallback =
285 ex::stop_callback_for_t<OuterStopToken,
286 stop_forwarder<SharedStateType>>;
287
288 SharedStateType shared_state;
289 F user_func;
290 std::optional<StopCallback> stop_callback;
291
292 union inner_storage_t {
293 inner_storage_t() {}
294 ~inner_storage_t() {}
295 InnerOpStateType op;
296 };
297 inner_storage_t inner_storage{};
298 bool started{false};
299
300 call_cc_op_state(OuterReceiver rcvr, F func)
301 : shared_state(std::move(rcvr)), user_func(std::move(func)) {}
302
303 call_cc_op_state(const call_cc_op_state &) = delete;
304 call_cc_op_state(call_cc_op_state &&) = delete;
305 call_cc_op_state &operator=(const call_cc_op_state &) = delete;
306 call_cc_op_state &operator=(call_cc_op_state &&) = delete;
307
308 ~call_cc_op_state() {
309 if (started) {
310 inner_storage.op.~InnerOpStateType();
311 }
312 }
313
314 void start() & noexcept {
315 // Upward cancellation: if the outer token is triggered, request stop on
316 // our local source so the inner work is cancelled. Registered before
317 // the inner op starts; if the outer token is already stopped the
318 // callback fires immediately here.
319 auto outer_token =
320 ex::get_stop_token(ex::get_env(shared_state.outer_receiver));
321 stop_callback.emplace(outer_token,
322 stop_forwarder<SharedStateType>{&shared_state});
323
324 escape_factory<ValueType> factory{&shared_state};
325 try {
326 ::new (&inner_storage.op) InnerOpStateType(ex::connect(
327 user_func(factory), InnerReceiverType{&shared_state}));
328 } catch (...) {
329 // A throwing user factory or connect must be reported through the
330 // error channel, not escape this noexcept start() and terminate.
331 ex::set_error(std::move(shared_state.outer_receiver),
332 std::current_exception());
333 return;
334 }
335 started = true;
336 ex::start(inner_storage.op);
337 }
338};

340template <class F, class ValueType>
341struct call_cc_sender {
342 using sender_concept = ex::sender_tag;
343
344 F user_func;
345
346 // The block's completions are derived from the inner sender (seen through
347 // the local stop-token environment), unioned with set_value_t(ValueType)
348 // for escape delivery and set_error_t(exception_ptr) for a throwing
349 // factory/connect.
350 template <typename, typename... Env>
351 static consteval auto get_completion_signatures() noexcept {
352 using inner_sender =
353 std::invoke_result_t<F &, escape_factory<ValueType>>;
354 using extra =
355 ex::completion_signatures<ex::set_value_t(ValueType),
356 ex::set_error_t(std::exception_ptr)>;
357 if constexpr (sizeof...(Env) == 0) {
358 using inner_sigs = ex::completion_signatures_of_t<inner_sender>;
359 return ex::detail::meta::unique<
360 ex::detail::meta::combine<inner_sigs, extra>>{};
361 } else {
362 using inner_sigs =
363 ex::completion_signatures_of_t<inner_sender,
364 inner_env_t<Env>...>;
365 return ex::detail::meta::unique<
366 ex::detail::meta::combine<inner_sigs, extra>>{};
367 }
368 }
369
370 template <ex::receiver OuterReceiver>
371 auto connect(OuterReceiver rcvr)
372 && -> call_cc_op_state<OuterReceiver, F, ValueType> {
373 return {std::move(rcvr), std::move(user_func)};
374 }
375
376 auto get_env() const noexcept -> ex::env<> { return {}; }
377};

417// escape_fn<V> — a public alias for the escape factory, so users can annotate
418// lambda parameters for call_cc_from deduction.
419template <class ValueType>
420using escape_fn = callcc_detail::escape_factory<ValueType>;

422// ---------------------------------------------------------------------------
423// call_cc<ValueType>(f) — the sender factory. `f` is invoked with an escape
424// factory; calling that factory with a ValueType produces an escape sender
425// which, when started, performs the early exit. ValueType is explicit because
426// the escape factory's signature must be known before `f` is invoked.
427// ---------------------------------------------------------------------------
428template <class ValueType, class F>
429auto call_cc(F func) -> callcc_detail::call_cc_sender<F, ValueType> {
430 return {std::move(func)};
431}
432
433// ---------------------------------------------------------------------------
434// call_cc_void(f) — convenience for early-exit with no meaningful value.
435// The escape factory's operator() takes no arguments; the underlying value
436// type is std::monostate.
437// ---------------------------------------------------------------------------
438template <class F>
439auto call_cc_void(F func) {
440 return call_cc<std::monostate>(
441 [f = std::move(func)](
442 callcc_detail::escape_factory<std::monostate> esc) {
443 return f(callcc_detail::void_escape_factory{esc});
444 });
445}
446
447// ---------------------------------------------------------------------------
448// call_cc_from(f) — deduces ValueType from the callable's parameter type.
449// Requires a non-generic lambda with a single escape_fn<V> parameter.
450// ---------------------------------------------------------------------------
451template <class F>
452auto call_cc_from(F func) {
453 using ValueType = callcc_detail::deduce_value_type_t<F>;
454 return call_cc<ValueType>(std::move(func));
455}
